// 1. 初始化配置（在main.js或入口文件中执行一次） import PortalSign from '@/utils/portal-sign'; PortalSign.init({ secret: 'your-secret-key', timestampWindow: 300, // 5分钟有效窗口 debug: true // 开发环境开启调试 }); // 2. 手动签名（适用于单个请求） const signResult = PortalSign.sign('POST', '/open/message/submit', JSON.stringify(data)); // 3. 发送请求时带上签名头 fetch('http://localhost:8081/open/message/submit', { method: 'POST', headers: { 'Content-Type': 'application/json', ...signResult.headers // 展开签名头 }, body: JSON.stringify(data) }).then(res => res.json());
// 在 main.js 中配置全局拦截器 import { createPortalService } from '@/utils/portal-sign-axios'; // 创建带自动签名功能的Axios实例 const portalApi = createPortalService({ baseURL: process.env.VUE_APP_PORTAL_SERVICE_URL || 'http://localhost:8081', }); // 之后所有请求都会自动签名！无需手动处理 portalApi.post('/open/message/submit', { name: '张三', contact: '13800138000', content: '测试留言' }).then(response => { console.log('请求成功！', response.data); });
